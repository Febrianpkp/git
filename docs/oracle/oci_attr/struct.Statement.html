Struct oracle::oci_attr::Statement[−][src]

pub struct Statement;
Expand description

A type parameter for Statement::oci_attr to get OCI_ATTR_STATEMENT, which is the text of the SQL statement prepared.

Examples

use oracle::oci_attr::Statement;

let mut stmt = conn.statement("select * from dual").build()?;
assert_eq!(stmt.oci_attr::<Statement>()?, "select * from dual");
